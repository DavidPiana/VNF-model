# 3. Importa la libreria
from amplpy import AMPL, modules

# 4. Carica i file nel percorso di sistema (questo RISOLVE l'errore "/x-ampl")
modules.load()

# 5. Attiva la tua licenza
try:
    modules.activate("43f778da-df2f-4eb6-a46c-03e3ed6d9df5")
    print("Licenza attivata con successo.")
except Exception as e:
    print(f"Attenzione, problema con la licenza: {e}")

# 6. Crea l'oggetto AMPL (ora funzionerà perché modules.load() ha preparato il terreno)
ampl = AMPL()

# 7. Imposta Gurobi come solutore
ampl.set_option("solver", "gurobi")

print("\n✅ Ambiente AMPL configurato correttamente e pronto all'uso!")
